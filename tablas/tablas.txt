CREATE TABLE usuarios (
    id SERIAL PRIMARY KEY,
    nombre VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL
);

CREATE TABLE IF NOT EXISTS predios (
    id SERIAL PRIMARY KEY,
    codigo VARCHAR(20) NOT NULL UNIQUE,
    nombre VARCHAR(100) DEFAULT '',
    area_hectareas FLOAT DEFAULT 0,
    tipo_sueldo VARCHAR(30) DEFAULT '',
    capacidad_maxima INT DEFAULT 0,
    observaciones TEXT DEFAULT '',
	activo int default 1
);

CREATE TABLE IF NOT EXISTS potreros (
    id SERIAL PRIMARY KEY,
id_predio int,
    codigo VARCHAR(20) NOT NULL,
    nombre VARCHAR(100) DEFAULT '',
    area_hectareas FLOAT DEFAULT 0,
    tipo_sueldo VARCHAR(30) DEFAULT '',
    capacidad_maxima INT DEFAULT 0,
    observaciones TEXT DEFAULT '',
	fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
	activo int default 1,
	CONSTRAINT fk_predios_potreros FOREIGN KEY (id_predio) REFERENCES predios(id),
	CONSTRAINT unique_potreros UNIQUE(id_predio,codigo)
);

CREATE TABLE IF NOT EXISTS sectores (
    id SERIAL PRIMARY KEY,
	id_predio int,
    codigo VARCHAR(20) NOT NULL,
    nombre VARCHAR(100) DEFAULT '',
	activo int default 1,
	CONSTRAINT fk_predios_sectores FOREIGN KEY (id_predio) REFERENCES predios(id),
	CONSTRAINT unique_sectores UNIQUE(id_predio,codigo)
);

CREATE TABLE IF NOT EXISTS lotes (
    id SERIAL PRIMARY KEY,
	id_predio int,
    codigo VARCHAR(20) NOT NULL,
    nombre VARCHAR(100) DEFAULT '',
	activo int default 1,
	CONSTRAINT fk_predios_lotes FOREIGN KEY (id_predio) REFERENCES predios(id),
	CONSTRAINT unique_lotes UNIQUE(id_predio,codigo)
);

CREATE TABLE IF NOT EXISTS grupos (
    id SERIAL PRIMARY KEY,
	id_predio int,
    codigo VARCHAR(20) NOT NULL,
    nombre VARCHAR(100) DEFAULT '',
	activo int default 1,
	CONSTRAINT fk_predios_grupos FOREIGN KEY (id_predio) REFERENCES predios(id),
	CONSTRAINT unique_grupos UNIQUE(id_predio,codigo)
);

CREATE TABLE IF NOT EXISTS diagnosticos_veterinarios (
    id SERIAL PRIMARY KEY,
	id_predio int,
	codigo VARCHAR(20) NOT NULL,
	nombre varchar(50) NOT NULL default '',
	tipo_diagnostico int not null default 0,
	comentarios text default '',
	activo int default 1,
	CONSTRAINT fk_predios_diagnosticos_veterinarios FOREIGN KEY (id_predio) REFERENCES predios(id),
	CONSTRAINT unique_diagnosticos_veterinarios UNIQUE(id_predio,codigo)
);

CREATE TABLE IF NOT EXISTS motivos_ventas (
    id SERIAL PRIMARY KEY,

CREATE TABLE IF NOT EXISTS causas_muerte (

CREATE TABLE IF NOT EXISTS imagenes (
    id SERIAL PRIMARY KEY,
	id_predio int,
	nombre_archivo varchar(100) not null,
	ruta_almacenamiento text not null,
	fecha_subida,
	comentarios text default '',
	CONSTRAINT unique_productos_imagenes UNIQUE(id_producto,id_imagen)
);

CREATE TABLE imagenes (
    id SERIAL PRIMARY KEY, 
	id_predio int,
    nombre_archivo VARCHAR(255) NOT NULL,
    ruta_archivo TEXT NOT NULL, 
    tipo_mime VARCHAR(50) NOT NULL,
    tamano BIGINT NOT NULL,
    fecha_creacion TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    descripcion TEXT,
	CONSTRAINT fk_predios_imagenes FOREIGN KEY (id_predio) REFERENCES predios(id),
	CONSTRAINT unique_imagenes UNIQUE(id_predio,nombre_archivo)
);

CREATE TABLE IF NOT EXISTS productos (
    id SERIAL PRIMARY KEY,
	id_predio int,
	codigo VARCHAR(20) NOT NULL,
	nombre varchar(50) NOT NULL default '',
	tipo_producto int default 0,
	comentarios text default '',
	activo int default 1,
	CONSTRAINT fk_predios_productos FOREIGN KEY (id_predio) REFERENCES predios(id),
	CONSTRAINT unique_productos UNIQUE(id_predio,codigo)
);

CREATE TABLE IF NOT EXISTS productos_imagenes (
    id SERIAL PRIMARY KEY,
	id_producto int not null,
	id_imagen int not null,
	CONSTRAINT fk_productos_imagenes1 FOREIGN KEY (id_producto) REFERENCES productos(id),
	CONSTRAINT fk_productos_imagenes2 FOREIGN KEY (id_imagen) REFERENCES imagenes(id),
	CONSTRAINT unique_productos_imagenes UNIQUE(id_producto,id_imagen)
);

CREATE TABLE IF NOT EXISTS veterinarios (
    id SERIAL PRIMARY KEY,
	id_predio int,
	codigo VARCHAR(20) NOT NULL,
	nombre varchar(50) NOT NULL default '',
	telefono varchar(50) NOT NULL default '',

	activo int default 1,

);

    id SERIAL PRIMARY KEY,
	id_predio int,

	CONSTRAINT unique_predios_parametros UNIQUE(id_predio,codigo)
);
